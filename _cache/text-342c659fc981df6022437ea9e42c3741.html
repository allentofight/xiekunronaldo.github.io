def cool_dude(arg1="Miles", arg2="Coltrane", arg3="Roach") 

    "#{arg1}, #{arg2}, #{arg3}."

end

cool_dude                           # => "Miles, Coltrane, Roach."

cool_dude("Bart")                   # => "Bart, Coltrane, Roach."

cool_dude("Bart", "Elwood")         # => "Bart, Elwood, Roach."

cool_dude("Bart", "Elwood", "Linus") # => "Bart, Coltrane, Linus."
