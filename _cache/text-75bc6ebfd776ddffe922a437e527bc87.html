def price_in_cents=(cents) 

    @price = cents / 100.0

end

book = BookInStock.new("isbn1", 33.80)

puts "Price = #{book.price}"

puts "Price in cents = #{book.price_in_cents}" 


book.price_in_cents = 1234


puts "Price = #{book.price}"


puts "Price in cents = #{book.price_in_cents}"


produces:


Price          = 33.8


Price in cents = 3380


Price          = 12.34


Price in cents = 1234
