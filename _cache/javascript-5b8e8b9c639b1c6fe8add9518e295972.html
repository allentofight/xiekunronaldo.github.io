Ext.define('Chapter4Data.model.Person', {
     extend: 'Ext.data.Model',
     config: {
         fields: [{
             name: 'entryId',
             type: 'int'
         }, {
             name: 'firstName',
             type: 'string'
         }, {
             name: 'lastName',
             type: 'string'
         }, {
             //
             // ... snip ...
             //
         }, {
             name: 'enabled',
             type: 'boolean',
             defaultValue: true
         }, {
             name: 'createdOn',
             type: 'date'
         }],
         idProperty: 'entryId',
         belongsTo: 'Company',
         hasMany: {
             model: 'Orders',
             name: 'orders'
         },
         validations: [{
             type: 'presence',
             field: 'firstName'
         }, {
             type: 'presence',
             field: 'lastName'
         }, {
             type: 'length',
             field: 'description',
             min: 15
         }, {
             type: 'inclusion',
             field: 'role',
             list: ['User', 'Admin']
         }]
     }
 });
