def five(a, b, c, d, e)

    "I was passed #{a} #{b} #{c} #{d} #{e}"

end

five(1,2,3,4,5) #=>"Iwaspassed12345" 

five(1, 2, 3, *['a', 'b']) # => "I was passed 1 2 3 a b" 

five(*['a', 'b'], 1, 2, 3) # => "I was passed a b 1 2 3" five(*(10..14)) 

#=>"Iwaspassed1011121314" 

five(*[1,2], 3, *(4..5)) #=>"Iwaspassed12345"
