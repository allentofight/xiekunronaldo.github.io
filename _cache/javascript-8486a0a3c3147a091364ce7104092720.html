{
        xtype: 'button',
        text: 'Touch to open panel',
        itemId: 'openPanelButton',
        listeners: {
            tap: function (button, e, eOpts) { 
                if (!button.panel) {
                    button.panel = Ext.widget('panel', {
                        html: 'This is a panel',
                        padding: 20,
                        width: 280,
                        modal: true, 
                        hideOnMaskTap: true, // 
                        hidden: true, 
                        hideAnimation: 'fadeOut', 
                        showAnimation: 'fadeIn'
                }); 
            }
                button.panel.showBy(button);
            }
        } 
}
