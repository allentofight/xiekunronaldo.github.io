{
    xtype: 'button',
    text: 'Touch to show sheet',
    itemId: 'openButton',
    listeners: {
        tap: function(button, e, eOpts) {
            if (!button.sheet) {
                button.sheet = Ext.widget('sheet', {
                    html: 'Some text inside of the sheet',
                    style: 'color: white; font-weight: bold',
                    stretchX: true, //flag
                    hideOnMaskTap: true
                });
                Ext.Viewport.add(button.sheet);
            }
            button.sheet.show();
        }
    }
}
