class BookInStock

    attr_writer :isbn, :price

    def initialize(isbn, price) 

        @isbn = isbn

        @price = Float(price)

    end

    def price=(new_price) 

        @price = new_price

    end

# ...

end



book = BookInStock.new("isbn1", 33.80) 

puts "ISBN = #{book.isbn}"

puts "Price = #{book.price}" 

book.price = book.price * 0.75

puts "New price = #{book.price}"
