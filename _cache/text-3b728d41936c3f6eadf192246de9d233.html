class SomeClass 

    def []=(*params)

        value = params.pop

        puts "Indexed with #{params.join(', ')}" 

        puts "value = #{value.inspect}"

    end

end

s = SomeClass.new

s[1] = 2

s['cat', 'dog'] = 'enemies'

produces:

       Indexed with 1

       value = 2

       Indexed with cat, dog

       value = "enemies"
